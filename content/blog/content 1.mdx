Server Components shine when you combine `cache()` with `suspense`. Here's a skeleton we reuse:

```tsx
import { Suspense } from "react";

async function Charts() {
	const [kpis, metrics] = await Promise.all([getKpis(), getMetrics()]);
	return <ChartGrid kpis={kpis} metrics={metrics} />;
}

export default function Dashboard() {
	return (
		<Suspense fallback={<Skeleton />}> 
			<Charts />
		</Suspense>
	);
}
```

Remember: GFM tables plus MDX lets you spec API payloads inline with the components.
